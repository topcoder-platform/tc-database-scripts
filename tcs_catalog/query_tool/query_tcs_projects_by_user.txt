SELECT {+AVOID_FULL(p)}
       DISTINCT p.project_id, p.project_status_id, p.project_category_id,
       p.create_user, p.create_date, p.modify_user, p.modify_date
FROM project p
INNER JOIN resource r ON r.project_id=p.project_id
INNER JOIN resource_info ri ON ri.resource_id=r.resource_id AND ri.resource_info_type_id=1 AND ri.value = '@uid@'
WHERE p.project_status_id = 1
